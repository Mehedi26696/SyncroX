//LANG:c
#include <stdio.h>
#include <stdlib.h>

int gcd(int a, int b) {
    a = abs(a);
    b = abs(b);
    while (b != 0) {
        int temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

int main(void) {
    int a, b;
    // e.g. input: 2 3
    if (scanf("%d %d", &a, &b) != 2) return 0;

    printf("%d\n", gcd(a, b));
    return 0;
}
